from os import path
from bs4 import BeautifulSoup
from pelican import signals, readers, contents


def extract_toc(content):
    if isinstance(content, contents.Static):
        return
    soup = BeautifulSoup(content._content,'html.parser')
    filename = content.source_path
    extension = path.splitext(filename)[1][1:]
    toc = ''
    toc_lis = []
    # if it is a Markdown file
    if extension in readers.MarkdownReader.file_extensions:
        toc = soup.find('div', class_='toc')
    # else if it is a reST file
    elif extension in readers.RstReader.file_extensions:
        toc = soup.find('div', class_='contents topic')
    if toc:
        toc.extract()
        content._content = soup.decode()
        content.toc = toc.decode()

        def direct_lis(uls):
            uls = uls.find('ul')
            lis = []
            for li in uls.find_all('li', recursive=False):
                child = []
                name = li.find('a').get_text().decode()
                link = li.find('a').get('href').decode()
                if li.find('ul'):
                    child = direct_lis(li)
                lis.append((name, link, child))
            return lis

        def all_lis(uls):
            lis = []
            for li in uls.find_all('li'):
                child = []
                name = li.find('a').get_text().decode()
                link = li.find('a').get('href').decode()
                lis.append((name, link, child))
            return lis

        content.toc_lis = direct_lis(toc)
        #content.toc_lis = all_lis(toc)

def register():
    signals.content_object_init.connect(extract_toc)
